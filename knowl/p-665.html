\begin{align*} A \adj A \amp = \begin{cases} \det A \amp \textrm{if } i=j\\ 0 \amp \textrm{otherwise} \end{cases}\\ \amp = (\det A)I \end{align*}

\begin{equation*} A\frac1{\det A} \adj A=I \end{equation*}
