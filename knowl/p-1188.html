\begin{equation*} \begin{array}{rl} L(\vec x) \amp =L(x_1\vec e_1 + x_2\vec e_2+\cdots+x_m\vec e_m)\\ \amp =x_1L(\vec e_1) + x_2L(\vec e_2)+\cdots+x_mL(\vec e_m)\\ \amp =x_1\vec f_1 + x_2\vec f_2+\cdots+x_n\vec f_m \end{array} \end{equation*}
