\begin{gather*} (b_{i,1}+c_{i,1})a_{1,j} + (b_{i,2}+c_{i,2})a_{2,j}+ \cdots+ (b_{i,n}+c_{i,n})a_{n,j}\\ = b_{i,1}a_{1,j}+c_{i,1}a_{1,j} +\cdots + b_{i,n}a_{1,n}+c_{i,n}a_{n,j} \text{.} \end{gather*}
